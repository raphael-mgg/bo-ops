A SOLUÇÃO · MACRO VBA

Uma macro varre a pasta, identifica a corretora pelo nome do arquivo (Select Case), aplica o parser correto pra cada layout, e despeja tudo normalizado na aba Boletas_Corretoras. 30 segundos no lugar de 30 minutos.

macros_backoffice.bas VBA 
'==============================================================================
' MACRO 1 — IMPORTAR BOLETAS DAS CORRETORAS
' Lê arquivos da pasta configurada e popula a aba Boletas_Corretoras.
' Cada corretora tem layout próprio, por isso há um Select Case.
'==============================================================================
Public Sub ImportarBoletasCorretoras()
    Dim wsDest As Worksheet
    Dim caminho As String, arquivo As String
    Dim linhaDest As Long
    Dim t0 As Single: t0 = Timer

    Set wsDest = ThisWorkbook.Sheets("Boletas_Corretoras")
    LimparAba wsDest
    linhaDest = 2

    caminho = LerParametro("PASTA_CORRETORAS")
    If Right(caminho, 1) <> "\" Then caminho = caminho & "\"

    arquivo = Dir(caminho & "*.*")
    If arquivo = "" Then
        MsgBox "Nenhum arquivo encontrado em " & caminho, vbExclamation
        Exit Sub
    End If

    Application.ScreenUpdating = False

    Do While arquivo <> ""
        ' identifica corretora pelo nome do arquivo
        Select Case True
            Case InStr(UCase(arquivo), "XP_") > 0
                linhaDest = ImportarXP(caminho & arquivo, wsDest, linhaDest)
            Case InStr(UCase(arquivo), "BTG_") > 0
                linhaDest = ImportarBTG(caminho & arquivo, wsDest, linhaDest)
            Case InStr(UCase(arquivo), "ITAU_BBA_") > 0
                linhaDest = ImportarItauBBA(caminho & arquivo, wsDest, linhaDest)
            Case InStr(UCase(arquivo), "BRADESCO_") > 0
                linhaDest = ImportarBradesco(caminho & arquivo, wsDest, linhaDest)
        End Select
        arquivo = Dir
    Loop

    Application.ScreenUpdating = True

    MsgBox "Importação concluída!" & vbCrLf & _
           (linhaDest - 2) & " boletas importadas em " & _
           Format(Timer - t0, "0.00") & "s.", vbInformation
End Sub

' ----- Parser específico da XP (delimitador ; + vírgula decimal) --------
Private Function ImportarXP(arq As String, wsDest As Worksheet, _
                              linhaInicial As Long) As Long
    Dim fNum As Integer, linha As String
    Dim partes() As String
    Dim ehHeader As Boolean: ehHeader = True
    Dim r As Long: r = linhaInicial

    fNum = FreeFile
    Open arq For Input As #fNum
    Do While Not EOF(fNum)
        Line Input #fNum, linha
        If ehHeader Then ehHeader = False: GoTo prox
        partes = Split(linha, ";")
        If UBound(partes) >= 7 Then
            wsDest.Cells(r, 1).Value = "XP_" & Dir(arq)
            wsDest.Cells(r, 2).Value = "XP"
            wsDest.Cells(r, 3).Value = CLng(partes(7))                ' boleta_id
            wsDest.Cells(r, 4).Value = Left(partes(0), 4) & "-" & _
                                       Mid(partes(0), 5, 2) & "-" & _
                                       Right(partes(0), 2)            ' data
            wsDest.Cells(r, 5).Value = partes(1)                       ' fundo
            wsDest.Cells(r, 6).Value = partes(2)                       ' ativo
            wsDest.Cells(r, 7).Value = partes(3)                       ' lado
            wsDest.Cells(r, 8).Value = CLng(partes(4))                 ' qtd
            wsDest.Cells(r, 9).Value = StrParaNumero(partes(5))        ' preço
            wsDest.Cells(r, 10).Value = StrParaNumero(partes(6))       ' financeiro
            r = r + 1
        End If
prox:
    Loop
    Close #fNum
    ImportarXP = r
End Function

' Helper: converte "12,34" → 12.34
Private Function StrParaNumero(ByVal s As String) As Double
    s = Replace(s, ".", "")
    s = Replace(s, ",", ".")
    If IsNumeric(s) Then StrParaNumero = CDbl(s) Else StrParaNumero = 0
End Function

'==============================================================================
' MACRO 2 — CASAR OPERAÇÕES (mesa x corretoras)
' Compara cada boleta da mesa com a correspondente da corretora pelo boleta_id.
' Marca status: OK / DIVERGENTE / FALTA_CORRETORA / FALTA_MESA
'==============================================================================
Public Sub CasarOperacoes()
    Dim wsMesa As Worksheet, wsCorr As Worksheet, wsCas As Worksheet
    Dim ultMesa As Long, ultCorr As Long, r As Long, i As Long, j As Long
    Dim tolPreco As Double, tolQtd As Double
    Dim achou As Boolean
    Dim qtdM As Long, qtdC As Long
    Dim prM As Double, prC As Double, diffPct As Double
    Dim status As String, obs As String

    Set wsMesa = ThisWorkbook.Sheets("Boletas_Mesa")
    Set wsCorr = ThisWorkbook.Sheets("Boletas_Corretoras")
    Set wsCas = ThisWorkbook.Sheets("Casamento")
    LimparAba wsCas

    tolPreco = LerParametro("TOLERANCIA_PRECO")
    tolQtd = LerParametro("TOLERANCIA_QTD")

    ultMesa = wsMesa.Cells(wsMesa.Rows.Count, 1).End(xlUp).Row
    ultCorr = wsCorr.Cells(wsCorr.Rows.Count, 1).End(xlUp).Row
    r = 2
    Application.ScreenUpdating = False

    ' Para cada boleta da mesa, procurar a correspondente
    For i = 2 To ultMesa
        achou = False
        qtdM = wsMesa.Cells(i, 6).Value
        prM = wsMesa.Cells(i, 7).Value
        For j = 2 To ultCorr
            If wsMesa.Cells(i, 1).Value = wsCorr.Cells(j, 3).Value Then
                achou = True
                qtdC = wsCorr.Cells(j, 8).Value
                prC = wsCorr.Cells(j, 9).Value
                diffPct = IIf(prM = 0, 0, (prC - prM) / prM)

                If Abs(qtdM - qtdC) <= tolQtd And Abs(diffPct) <= tolPreco Then
                    status = "OK": obs = ""
                Else
                    status = "DIVERGENTE": obs = ""
                    If Abs(qtdM - qtdC) > tolQtd Then obs = obs & "qtd; "
                    If Abs(diffPct) > tolPreco Then obs = obs & "preco; "
                End If

                EscreverLinhaCasamento wsCas, r, wsMesa.Cells(i, 1).Value, _
                    wsMesa.Cells(i, 3).Value, wsMesa.Cells(i, 4).Value, _
                    wsMesa.Cells(i, 5).Value, qtdM, qtdC, prM, prC, _
                    qtdM - qtdC, diffPct, status, obs
                r = r + 1
                Exit For
            End If
        Next j

        If Not achou Then
            EscreverLinhaCasamento wsCas, r, wsMesa.Cells(i, 1).Value, _
                wsMesa.Cells(i, 3).Value, wsMesa.Cells(i, 4).Value, _
                wsMesa.Cells(i, 5).Value, qtdM, 0, prM, 0, qtdM, 0, _
                "FALTA_CORRETORA", "sem correspondente na corretora"
            r = r + 1
        End If
    Next i

    PintarCasamento wsCas, r - 1
    Application.ScreenUpdating = True
    MsgBox "Casamento concluído. " & (r - 2) & " linhas processadas.", vbInformation
End Sub

' Pinta as linhas conforme o status (verde/vermelho/amarelo)
Private Sub PintarCasamento(ws As Worksheet, ultLinha As Long)
    Dim i As Long, st As String
    For i = 2 To ultLinha
        st = ws.Cells(i, 11).Value
        Select Case st
            Case "OK"
                ws.Range(ws.Cells(i, 1), ws.Cells(i, 12)).Interior.Color = RGB(198, 239, 206)
            Case "DIVERGENTE"
                ws.Range(ws.Cells(i, 1), ws.Cells(i, 12)).Interior.Color = RGB(255, 199, 206)
            Case "FALTA_CORRETORA", "FALTA_MESA"
                ws.Range(ws.Cells(i, 1), ws.Cells(i, 12)).Interior.Color = RGB(255, 235, 156)
        End Select
    Next i
End Sub

Tem caixa pra liquidar tudo até as 16h?

A DOR REAL

Pra cada fundo, é preciso somar quanto sai (compras) e quanto entra (vendas) em cada data de liquidação. '==============================================================================
' MACRO 3 — GERAR NECESSIDADE DE CAIXA
' Agrupa boletas por fundo + data de liquidação. Compras saem caixa,
' vendas entram. Saldo líquido < 0 = precisa providenciar caixa.
'==============================================================================
Public Sub GerarNecessidadeCaixa()
    Dim wsMesa As Worksheet, wsCx As Worksheet
    Dim ult As Long, i As Long, r As Long
    Dim chave As String
    Dim dict As Object
    Dim fundo As String, dt As String, lado As String
    Dim valor As Double

    Set wsMesa = ThisWorkbook.Sheets("Boletas_Mesa")
    Set wsCx = ThisWorkbook.Sheets("Necessidade_Caixa")
    Set dict = CreateObject("Scripting.Dictionary")
    LimparAba wsCx

    ult = wsMesa.Cells(wsMesa.Rows.Count, 1).End(xlUp).Row

    ' Agrega compras e vendas por (fundo, data_liquidação)
    For i = 2 To ult
        fundo = wsMesa.Cells(i, 3).Value
        dt = CStr(wsMesa.Cells(i, 11).Value)
        lado = wsMesa.Cells(i, 5).Value
        valor = wsMesa.Cells(i, 8).Value
        chave = fundo & "|" & dt
        If Not dict.Exists(chave) Then dict.Add chave, Array(0, 0)
        Dim arr As Variant: arr = dict(chave)
        If lado = "C" Then arr(0) = arr(0) + valor Else arr(1) = arr(1) + valor
        dict(chave) = arr
    Next i

    ' Escreve resultado, com SHORTFALL em vermelho
    r = 2
    Dim k As Variant
    For Each k In dict.Keys
        Dim p() As String: p = Split(CStr(k), "|")
        Dim arr2 As Variant: arr2 = dict(k)
        wsCx.Cells(r, 1).Value = p(0)                       ' fundo
        wsCx.Cells(r, 2).Value = p(1)                       ' data
        wsCx.Cells(r, 3).Value = arr2(0)                    ' compras
        wsCx.Cells(r, 4).Value = arr2(1)                    ' vendas
        wsCx.Cells(r, 5).Value = arr2(1) - arr2(0)          ' saldo
        wsCx.Cells(r, 6).Value = IIf(arr2(1) - arr2(0) < 0, "SHORTFALL", "OK")
        If wsCx.Cells(r, 6).Value = "SHORTFALL" Then
            wsCx.Range(wsCx.Cells(r, 1), wsCx.Cells(r, 6)).Interior.Color = RGB(255, 199, 206)
        End If
        r = r + 1
    Next k

    MsgBox "Necessidade de caixa calculada para " & (r - 2) & _
           " combinações fundo/data.", vbInformation
End Sub

"""
Conciliação com custódia: cruza posição interna calculada
(posicao_inicial + boletas) com a posição enviada pela custódia.
"""
import pandas as pd
import numpy as np
from pathlib import Path

BASE = Path(__file__).resolve().parent.parent

def calcular_posicao_interna(pos_inicial, boletas):
    # Aplica boletas (compras +, vendas -) sobre a posição inicial
    mov = boletas.copy()
    mov["delta_qtd"] = np.where(
        mov["lado"] == "C",
        mov["quantidade"],
        -mov["quantidade"]
    )

    mov_agg = (mov.groupby(["fundo", "ativo"], as_index=False)
                  .agg(delta_qtd=("delta_qtd", "sum")))

    pos = pos_inicial.merge(mov_agg, on=["fundo", "ativo"], how="outer")
    pos["quantidade"] = pos["quantidade"].fillna(0)
    pos["delta_qtd"] = pos["delta_qtd"].fillna(0)
    pos["qtd_interno"] = pos["quantidade"] + pos["delta_qtd"]

    return pos[pos["qtd_interno"] != 0][["fundo", "ativo", "qtd_interno"]]


def conciliar_custodia(pos_interna, custodia):
    # Cruza posição interna x custódia. Marca o status da divergência.
    cust = custodia[["fundo", "ativo", "qtd_custodia"]].copy()

    conc = pos_interna.merge(
        cust,
        on=["fundo", "ativo"],
        how="outer",
        indicator=True
    )
    conc["qtd_interno"] = conc["qtd_interno"].fillna(0)
    conc["qtd_custodia"] = conc["qtd_custodia"].fillna(0)
    conc["diferenca"] = conc["qtd_interno"] - conc["qtd_custodia"]

    def classificar(row):
        if row["_merge"] == "left_only":
            return "SO_INTERNO"      # custódia não tem
        if row["_merge"] == "right_only":
            return "SO_CUSTODIA"     # interno não tem (evento corp.)
        if abs(row["diferenca"]) < 0.0001:
            return "OK"
        return "DIVERGENTE"

    conc["status_conciliacao"] = conc.apply(classificar, axis=1)
    conc = conc.drop(columns="_merge")

    qtd_ok = (conc["status_conciliacao"] == "OK").sum()
    qtd_div = (conc["status_conciliacao"] != "OK").sum()
    print(f"      ✓ {qtd_ok} posições OK, {qtd_div} com divergência")

    return conc


# --- USO ---
pos_inicial = pd.read_csv(BASE / "dados/posicao_inicial.csv", sep=";")
boletas = pd.read_csv(BASE / "dados/boletas_mesa.csv", sep=";")
custodia = pd.read_csv(BASE / "custodia/posicao_custodia_20260428.csv", sep=";")

    "CD_FUNDO": "fundo",

def marcar_mercado(pos_interna, precos, boletas, pos_inicial):
    """
    Aplica preço de fechamento sobre a quantidade pra obter valor de mercado.
    Calcula também P&L do dia (preço fechamento - preço médio histórico).
    """
    p = precos.rename(columns={"preco_fechamento": "preco_mercado"})

    # Preço médio do dia (média ponderada das boletas)
    pmd = (boletas.groupby(["fundo", "ativo"])
                  .apply(lambda g: (g["preco"] * g["quantidade"]).sum()
                                    / g["quantidade"].sum())
                  .reset_index(name="preco_medio_dia"))

    mtm = pos_interna.merge(p[["ativo", "preco_mercado"]], on="ativo", how="left")
    mtm = mtm.merge(pmd, on=["fundo", "ativo"], how="left")
    mtm = mtm.merge(pos_inicial[["fundo", "ativo", "preco_medio"]],
                    on=["fundo", "ativo"], how="left")

    mtm["preco_mercado"] = mtm["preco_mercado"].fillna(0)
    mtm["valor_mercado"] = mtm["qtd_interno"] * mtm["preco_mercado"]

    # P&L = (preço atual - preço de referência) × qtd
    mtm["preco_referencia"] = mtm["preco_medio"].fillna(mtm["preco_medio_dia"])
    mtm["pnl_dia"] = (mtm["preco_mercado"] - mtm["preco_referencia"]) * mtm["qtd_interno"]
    mtm["pnl_dia"] = mtm["pnl_dia"].fillna(0)

    return mtm


def calcular_pl_e_cota(mtm, boletas, caixa_inicial, cotas_circulacao, cota_anterior):
    """
    PL = soma(valor_mercado) + caixa atualizado.
    Caixa atualizado = caixa inicial + vendas - compras.
    Cota = PL / qtd cotas em circulação.
    """
    boletas = boletas.copy()
    boletas["sinal"] = np.where(boletas["lado"] == "C", -1, 1)
    fluxo_caixa = (boletas.assign(fluxo=boletas["sinal"] * boletas["valor_financeiro"])
                          .groupby("fundo")["fluxo"].sum())

    ativos_fundo = mtm.groupby("fundo")["valor_mercado"].sum()

    pl = pd.DataFrame({"fundo": list(cota_anterior.keys())})
    pl["valor_ativos"]      = pl["fundo"].map(ativos_fundo).fillna(0)
    pl["caixa_inicial"]     = pl["fundo"].map(caixa_inicial)
    pl["fluxo_caixa_dia"]   = pl["fundo"].map(fluxo_caixa).fillna(0)
    pl["caixa_final"]       = pl["caixa_inicial"] + pl["fluxo_caixa_dia"]
    pl["pl_total"]          = pl["valor_ativos"] + pl["caixa_final"]
    pl["cotas_circulacao"]  = pl["fundo"].map(cotas_circulacao)
    pl["cota_hoje"]         = pl["pl_total"] / pl["cotas_circulacao"]
    pl["cota_ontem"]        = pl["fundo"].map(cota_anterior)
    pl["rentabilidade_dia"] = (pl["cota_hoje"] - pl["cota_ontem"]) / pl["cota_ontem"]

    return pl

Onde tudo dá errado

Quebra de boleta
Mesa registrou X, corretora registrou Y. Resolução: gravação de voice trade ou log do FIX.
Falha de liquidação
Fundo sem caixa pra honrar compra. Solução: resgate de DI, venda de LFT, linha overnight.
Falha de custódia
PL interno ≠ PL custódia. Causas: evento corporativo, BTC, boleta cancelada de um lado só.
Cota errada
Republica c/ cota correta. Se afetou cotista que aplicou/resgatou, gestora indeniza.

DATA·OPS · A MESA DE DADOS

Antes de a mesa abrir, os dados já trabalharam.

Enquanto o backoffice cuida do fluxo das operações, o time de dados cuida do fluxo da informação: coleta preços, taxas e carteiras de várias fontes, valida a integridade de tudo e entrega números prontos pra decisão. A missão cabe numa frase: nenhum número chega à mesa sem ser coletado, tratado e conferido. Um preço errado que passa despercebido vira cota errada, enquadramento errado, decisão errada.

O dia da mesa de dados da Alpha Capital segue o mesmo formato do Diário de Operações: uma linha do tempo, um problema concreto por hora, e a ferramenta certa pra resolver.

ETL · PYTHON

O arquivo chegou sujo. Como sempre.

A DOR REAL

Todo dia, antes das 8h, chegam os arquivos: preços da B3, taxas da ANBIMA, indexadores (CDI, IMA-B, Ibovespa) e as carteiras que o administrador fiduciário fecha. Nenhum vem pronto: data em formato brasileiro, número com vírgula decimal, campo vazio no meio. ETL — Extract, Transform, Load — é o nome do processo que extrai o arquivo cru, trata cada um desses problemas e carrega o resultado padronizado no banco. Sempre nessa ordem.

CRU como o arquivo chega
DT_PREGAO;TICKER;PRECO_ULT
28/04/2026;PETR4;38,52
28/04/2026;TAEE11;345,10
28/04/2026;WEGE3;
TRATADO como entra no banco
data_pregao  ticker   preco_fech
2026-04-28   PETR4    38.52
2026-04-28   TAEE11   345.10   ← passa no ETL; a CONFERÊNCIA pega
2026-04-28   WEGE3    (log de pendências — campo vazio)
A SOLUÇÃO · SCRIPT PYTHON

Repare no detalhe: o ETL não corrige o preço estranho da TAEE11 — tratar formato é papel do ETL; julgar se o número faz sentido é papel da conferência das 9h. Separar as duas coisas evita "correções" silenciosas que escondem erro de fonte.

etl_precos.py PYTHON · pandas 
"""
ETL diário — preços B3.
O arquivo chega 'sujo': data BR, decimal com vírgula, campo vazio.
Extract → Transform → Load, sempre nessa ordem.
"""
import pandas as pd
from sqlalchemy import create_engine

engine = create_engine("sqlite:///alpha_dados.db")

# ---- EXTRACT: lê tudo como texto cru (nenhuma conversão automática) ----
df = pd.read_csv("entrada/precos_b3_20260428.csv", sep=";", dtype=str)

# ---- TRANSFORM: cada problema do arquivo tem tratamento explícito ----
# 1) data em formato BR (28/04/2026) → ISO (2026-04-28)
df["data_pregao"] = pd.to_datetime(df["DT_PREGAO"], format="%d/%m/%Y").dt.date

# 2) número BR ("1.234,56") → float (1234.56)
df["preco_fech"] = pd.to_numeric(
    df["PRECO_ULT"].str.replace(".", "", regex=False)
                   .str.replace(",", ".", regex=False),
    errors="coerce")  # campo vazio vira NaN — tratado abaixo

# 3) campo vazio → NÃO inventa preço: separa, registra e segue
faltantes = df[df["preco_fech"].isna()]
if len(faltantes):
    faltantes.to_csv("log/pendencias_precos.csv", index=False)
    print(f"[aviso] {len(faltantes)} ativo(s) sem preço — vai pra conferência")
df = df.dropna(subset=["preco_fech"])

# ---- LOAD: grava a carga do dia na tabela de preços ----
df[["data_pregao", "TICKER", "preco_fech"]] \
  .rename(columns={"TICKER": "ticker"}) \
  .to_sql("fato_preco", engine, if_exists="append", index=False)

print(f"[ok] {len(df)} preços carregados em fato_preco")

A PARTE MAIS IMPORTANTE DO DIA

    

      Dado carregado não é dado confiável. Antes de liberar qualquer número pra mesa, rodam
      as três checagens clássicas de integridade — e enquanto houver alerta aberto,
      nada sai. É a versão de dados do "tudo bate" do backoffice.
    

  


  

    

      
a) COMPLETUDE

      
Todo fundo tem cota hoje? Todo ativo em carteira tem preço? Se falta linha, algo não chegou — e ninguém percebe até precisar.

    

    

      
b) SANIDADE

      
Algum preço variou mais de ±10% contra ontem? Outlier assim quase sempre é erro de feed — vírgula deslocada, fator de ajuste — e não mercado.

    

    

      
c) BATIMENTO

      
A cota oficial do administrador bate com o cálculo interno ("shadow NAV")? Tolerância: 1 basis point (0,01%). Acima disso, investiga.

POR QUE ISSO IMPORTA

  

  
Planilha manual não escala. Pipeline sim.


  

    

      Substituir planilhas manuais por pipelines estruturados não é estética: reduz erro
      operacional (cada tratamento é explícito e testável), libera tempo
      (a carga que tomava a manhã roda sozinha antes das 8h) e garante que o gestor decide com
      dado íntegro — coletado, tratado e conferido. A mesa de dados e o backoffice
      são as duas engrenagens da mesma máquina: uma garante que a operação bate, a outra que a
      informação bate.
    

    

      As consultas que o time de dados roda todo dia (completude por fundo, última data por tabela,
      batimento de cota) estão em sql/relatorios_dataops.sql no repositório do projeto.

      Comecei a estudar mercado financeiro como muita gente começa: lendo sobre ações, fundos,
      juros — sem entender exatamente como o sistema funciona por dentro.
    


    

      Quando passei a olhar com mais atenção o caminho de entrada no mercado, percebi um padrão:
      a maior parte dos estagiários e analistas júnior entra pela porta do backoffice.
      É a área que ensina o "encanamento" do mercado — boletas, liquidação, custódia, conciliação,
      fechamento de cota. Mas, ao mesmo tempo, é a área menos explicada nos cursos, nos livros,
      nos vídeos. Você só entende quando já está lá dentro.
    


    

      Construí este site pra reduzir essa distância. Cada página representa uma hora real do dia
      de quem trabalha na operação de uma gestora. Cada ferramenta interativa simula uma rotina
      concreta — importar boletas de corretoras, conciliar posições com a custódia, marcar a mercado,
      calcular cota, montar uma carteira no home broker. O código por trás de tudo está aberto:
      VBA, Python, DAX, JavaScript.
    


    

      A finalidade é dupla: didática, pra quem está começando ter um mapa antes
      de entrar; e interativa, porque ler sobre o trabalho não substitui experimentar
      o trabalho. Aprende-se mais errando uma boleta simulada do que lendo dez parágrafos sobre o
      conceito.
